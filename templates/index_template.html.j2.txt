(function(){
  'use strict';
  var toggle=document.querySelector('.nav-toggle'),navLinks=document.getElementById('main-nav');
  if(toggle&&navLinks){
    toggle.addEventListener('click',function(){var open=toggle.getAttribute('aria-expanded')==='true';toggle.setAttribute('aria-expanded',open?'false':'true');toggle.setAttribute('aria-label',open?'Open navigation menu':'Close navigation menu');navLinks.classList.toggle('is-open',!open)});
    navLinks.addEventListener('click',function(e){if(e.target.tagName==='A'){toggle.setAttribute('aria-expanded','false');toggle.setAttribute('aria-label','Open navigation menu');navLinks.classList.remove('is-open')}});
    document.addEventListener('click',function(e){if(!toggle.contains(e.target)&&!navLinks.contains(e.target)){toggle.setAttribute('aria-expanded','false');navLinks.classList.remove('is-open')}});
    document.addEventListener('keydown',function(e){if(e.key==='Escape'){toggle.setAttribute('aria-expanded','false');navLinks.classList.remove('is-open');toggle.focus()}});
  }
  var CK='eu_data_consent_v1',banner=document.getElementById('cookie-banner'),btnOk=document.getElementById('cb-accept'),btnNope=document.getElementById('cb-decline');
  function rc(){try{return localStorage.getItem(CK)}catch(e){return null}}
  function sc(v){try{localStorage.setItem(CK,v)}catch(e){}}
  function cb(){if(!banner)return;banner.style.transition='opacity .22s ease,transform .22s ease';banner.style.opacity='0';banner.style.transform='translateY(100%)';setTimeout(function(){banner.style.display='none';banner.classList.remove('is-visible')},240)}
  if(!rc()&&banner){setTimeout(function(){banner.classList.add('is-visible');banner.style.display='block';banner.style.opacity='';banner.style.transform=''},900)}
  if(btnOk)btnOk.addEventListener('click',function(){sc('accepted');cb()});
  if(btnNope)btnNope.addEventListener('click',function(){sc('declined');cb()});
  document.addEventListener('keydown',function(e){if(e.key==='Escape'&&banner&&banner.classList.contains('is-visible')){sc('declined');cb()}});
}());
